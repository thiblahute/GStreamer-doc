Example launch line

 # server:
 gst-launch-1.0 fdsrc fd=1 ! tcpserversink port=3000
 # client:
 gst-launch-1.0 tcpclientsrc port=3000 ! fdsink fd=2

Enumerations

GstFormat

undefined (0x00000000) – GST_FORMAT_UNDEFINED

default (0x00000001) – GST_FORMAT_DEFAULT

bytes (0x00000002) – GST_FORMAT_BYTES

time (0x00000003) – GST_FORMAT_TIME

buffers (0x00000004) – GST_FORMAT_BUFFERS

percent (0x00000005) – GST_FORMAT_PERCENT

GstMultiHandleSinkRecoverPolicy

none (0x00000000) – Do not try to recover

latest (0x00000001) – Resync client to latest buffer

soft-limit (0x00000002) – Resync client to soft limit

keyframe (0x00000003) – Resync client to most recent keyframe

GstMultiHandleSinkSyncMethod

latest (0x00000000) – Serve starting from the latest buffer

next-keyframe (0x00000001) – Serve starting from the next keyframe

latest-keyframe (0x00000002) – Serve everything since the latest keyframe (burst)

burst (0x00000003) – Serve burst-value data to client

burst-keyframe (0x00000004) – Serve burst-value data starting on a keyframe

burst-with-keyframe (0x00000005) – Serve burst-value data preferably starting on a keyframe
