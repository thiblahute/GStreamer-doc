Example launch line

 # server:
 gst-launch-1.0 fdsrc fd=1 ! tcpserversink port=3000
 # client:
 gst-launch-1.0 tcpclientsrc port=3000 ! fdsink fd=2

enum GstMultiHandleSinkRecoverPolicy

Name Description Value
none Do not try to recover 0x00000000
latest Resync client to latest buffer 0x00000001
soft-limit Resync client to soft limit 0x00000002
keyframe Resync client to most recent keyframe 0x00000003

enum GstMultiHandleSinkSyncMethod

Name Description Value
latest Serve starting from the latest buffer 0x00000000
next-keyframe Serve starting from the next keyframe 0x00000001
latest-keyframe Serve everything since the latest keyframe (burst) 0x00000002
burst Serve burst-value data to client 0x00000003
burst-keyframe Serve burst-value data starting on a keyframe 0x00000004
burst-with-keyframe Serve burst-value data preferably starting on a keyframe 0x00000005
